core.windowing

Pure math functions for viewport window calculations and navigation.

Navigation Helpers

navigate


def navigate(
    window_start:int, # Current first visible row
    direction:str, # 'up', 'down', 'page_up', 'page_down', 'first', 'last'
    visible_rows:int, # Number of visible rows
    total_items:int, # Total item count
)->int: # New window_start (clamped)

Compute new window_start for a navigation action.

navigate_cursor


def navigate_cursor(
    cursor_index:int, # Current cursor position (-1 treated as window_start)
    direction:str, # 'up' or 'down'
    window_start:int, # Current first visible row
    visible_rows:int, # Number of visible rows
    total_items:int, # Total item count
    is_skippable:Optional=None, # Predicate: index -> skip this item?
)->Tuple: # (new_cursor, new_window_start, window_changed)

Move cursor up/down within the visible window, scrolling at edges.

def find_nearest_focusable(
    index:int, # Starting index to search from
    total_items:int, # Total item count
    is_skippable:Optional=None, # Predicate: index -> skip?
    direction:int=1, # Search direction: 1 (forward) or -1 (backward)
)->int: # Nearest focusable index, or -1 if none found

Find the nearest non-skippable index from a starting position.

# Test find_nearest_focusable
skip_set = {0, 5, 10}
skip_fn = lambda idx: idx in skip_set

# No skippable predicate — returns clamped index
assert find_nearest_focusable(3, 20) == 3
assert find_nearest_focusable(-1, 20) == 0
assert find_nearest_focusable(25, 20) == 19

# Non-skippable index — returns as-is
assert find_nearest_focusable(3, 20, skip_fn) == 3

# Skippable index — searches forward first
assert find_nearest_focusable(0, 20, skip_fn, direction=1) == 1
assert find_nearest_focusable(5, 20, skip_fn, direction=1) == 6
assert find_nearest_focusable(10, 20, skip_fn, direction=1) == 11

# Skippable index — searches backward when direction=-1
assert find_nearest_focusable(5, 20, skip_fn, direction=-1) == 4
assert find_nearest_focusable(10, 20, skip_fn, direction=-1) == 9

# Skippable index at start, backward search falls back to forward
assert find_nearest_focusable(0, 20, skip_fn, direction=-1) == 1

# All items skippable — returns -1
assert find_nearest_focusable(0, 3, lambda idx: True) == -1

# Empty collection — returns -1
assert find_nearest_focusable(0, 0, skip_fn) == -1
